0032         %% shouldSwap(deltaCost,temp)
0033         function flag = shouldSwap(obj,deltaCost,temp)
0034             %calculates p(swap) based on deltaCost and temperature, using logistic function
0035             %
0036             % Currently implemented using OO design in anticipation of
0037             % possible expansion of the code, though this is not strictly
0038             % necessary at present.
0039             %
0040             %Parameters:
0041             %   deltaCost - diff in cost between curCost and swapCost
0042             %   temp - temperature
0043            
0044             % since we want to make swaps when cost is negative, the
0045             % negative sign that would usually proceed the net input
0046             % (delta cost) is removed in this version of the logistic fcn.
0047            p= 1/(1+exp(deltaCost*1/temp)); 
0048            
0049            if rand <= p               
0050                flag = 1;
0051            else
0052                flag = 0;
0053            end
0054         end
