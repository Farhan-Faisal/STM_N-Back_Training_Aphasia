0024 function setVerbosePrintVerbosity(fieldName,val)
0025     %% sets the verbosity for a particular field to manipulate whether and how it is output
0026    if (ischar(fieldName) == false) 
0027         error('{fieldName} supplied to setVerbosePrintVerbosity is not a String');
0028    end
0029    
0030    if(isinteger(val) == false)
0031        error('{val} supplied to setVerbosePrintVerbosity is not an integer');
0032    end
0033    
0034     global verbosityFlags;
0035     if(isempty(verbosityFlags))
0036         initVerbosePrint(); 
0037     end
0038     
0039     
0040     if (isfield(verbosityFlags,fieldName))
0041         setfield(verbosityFlags,fieldName) = val; %#ok<NASGU,NASGU: Stored in global for later use>
0042     else
0043        error('{fieldName} is not a registered verbosityFlag');
0044     end

0046 
0047 end
